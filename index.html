function updateBrightness() { const lumens = parseFloat(document.getElementById('lumens-display').textContent) || 0; const gain = parseFloat(document.getElementById('gain-brightness').value) || 1.0; const ambient = parseFloat(document.getElementById('ambient').value) || 0; const widthM = getImageWidthInMeters(); const heightM = getImageHeightInMeters(); const area = widthM * heightM; const illuminance = area > 0 ? (lumens / area) : 0; const effective = Math.max((illuminance * gain) / Math.PI, 5); // 1. Keep the formula calculation clean as a pure number first const rawContrastRatio = ambient > 0 ? ((illuminance * gain + ambient) / ambient) : 0; document.getElementById('area').textContent = area.toFixed(2); document.getElementById('illuminance').textContent = illuminance.toFixed(0); document.getElementById('effective').textContent = effective.toFixed(0); // 2. Set the DOM textual display separately using .toFixed() document.getElementById('contrast-ratio').textContent = ambient > 0 ? rawContrastRatio.toFixed(1) : "—"; let verdictText = ""; let color = "black"; // 3. Evaluate the condition using the PURE NUMBER variable (rawContrastRatio) if (rawContrastRatio > 10.4) { verdictText = "Excellent"; color = "green"; } else if (rawContrastRatio > 8.4) { verdictText = "Good"; color = "green"; } else if (rawContrastRatio > 5.4) { verdictText = "Acceptable"; color = "orange"; } else { verdictText = "Too Dim"; color = "red"; } const verdictEl = document.getElementById('verdict'); verdictEl.textContent = verdictText; verdictEl.style.color = color; }
